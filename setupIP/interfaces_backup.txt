# The loopback network interface
auto lo
iface lo inet loopback

auto eth0
iface eth0 inet static
        address 137.74.195.38
        netmask 255.255.255.255
        post-up /sbin/ip route add 137.74.192.1 dev eth0
        post-up /sbin/ip route add default via 137.74.192.1
        pre-down /sbin/ip route del default via 137.74.192.1
        pre-down /sbin/ip route del 137.74.192.1 dev eth0
        dns-nameserver 213.186.33.99
        dns-search ovh.net
